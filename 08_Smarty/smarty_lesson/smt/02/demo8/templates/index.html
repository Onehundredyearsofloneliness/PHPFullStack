---------foreach遍历数组 arr----------

<{/foreach}>
---------foreach遍历二维数组 brr----------
<{$brr}>
<{foreach $brr as $val1}> <{$val1["uName"]}>---<{$val1["uAge"]}>---<{$val1["uTel"]}>
<{/foreach}>
---------foreach遍历二维数组 brr $key=>$val----------
<{$brr}>
<{foreach $brr as $key=>$val1}> <{$key}> ： <{$val1["uName"]}>---<{$val1["uAge"]}>---<{$val1["uTel"]}>
<{/foreach}>
---------section-crr---------
<{$crr}>
<{section name=key loop=$crr}> <{$crr[key]}>
<{/section}>
---------section遍历二维数组 -drr---------
<{$drr}>
<{section name=key1 loop=$drr}> <{$drr[key1].uName}>
<{/section}>
